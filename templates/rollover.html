let _counts = {}; let _yearOrder = []; let _rollTarget = null; let _confirmPhrase = ''; function showStatus(msg, type='ok') { const bar = document.getElementById('status-bar'); bar.textContent = msg; bar.className = `status-bar status-${type}`; setTimeout(() => bar.className = 'status-bar hidden', 5000); } async function loadCounts() { try { const resp = await fetch('/api/rollover/counts'); const d = await resp.json(); if (!d.ok) { showStatus('Could not load: ' + d.error, 'error'); return; } _counts = d.counts; _yearOrder = d.year_order; render(); } catch(e) { showStatus('Request failed: ' + e.message, 'error'); } finally { document.getElementById('loading').style.display = 'none'; document.getElementById('year-grid').style.display = 'grid'; } } function render() { const grid = document.getElementById('year-grid'); grid.innerHTML = ''; _yearOrder.forEach(yr => { const info = _counts[yr]; if (!info) return; const isY6 = yr === '6'; const nextYr = isY6 ? null : String(parseInt(yr) + 1); const empty = info.total === 0; const clsRows = info.classes.map(c => { const pendingPill = c.pending ? `${c.pending} pending` : ''; return `
${c.id}${pendingPill}${c.count} pupil${c.count !== 1 ? 's' : ''}
`; }).join(''); const footerContent = empty ? `No pupils — nothing to roll` : `→ ${isY6 ? '🗄 Archive' : 'Y'+nextYr} `; const card = document.createElement('div'); card.className = 'year-card'; card.id = `card-y${yr}`; card.innerHTML = `
Year ${yr}${info.total} pupil${info.total !== 1 ? 's' : ''} total
${clsRows}
${footerContent}
`; grid.appendChild(card); }); } function openConfirm(yr) { _rollTarget = yr; const info = _counts[yr]; const isY6 = yr === '6'; const nextYr = isY6 ? null : String(parseInt(yr) + 1); _confirmPhrase = isY6 ? 'ARCHIVE Y6' : ('ROLL Y' + yr); document.getElementById('confirm-title').textContent = isY6 ? 'Archive Year 6' : ('Roll Year ' + yr + ' to Year ' + nextYr); document.getElementById('confirm-body').textContent = isY6 ? (info.total + ' Y6 pupils will be archived and removed from active classes.') : (info.total + ' Y' + yr + ' pupils will be moved to Year ' + nextYr + ' and marked as Pending. Use Class Manager afterwards to assign each pupil to their new class.'); document.getElementById('confirm-phrase-label').textContent = _confirmPhrase; document.getElementById('confirm-phrase-input').value = ''; document.getElementById('confirm-phrase-input').placeholder = _confirmPhrase; const btn = document.getElementById('confirm-btn'); btn.textContent = isY6 ? 'Archive Y6' : ('Roll Y' + yr + ' to Y' + nextYr); btn.disabled = true; btn.style.opacity = '.4'; btn.style.cursor = 'not-allowed'; document.getElementById('confirm-modal').style.display = 'flex'; setTimeout(() => document.getElementById('confirm-phrase-input').focus(), 80); } function checkConfirmPhrase() { const typed = document.getElementById('confirm-phrase-input').value.trim().toUpperCase(); const btn = document.getElementById('confirm-btn'); const match = typed === _confirmPhrase; btn.disabled = !match; btn.style.opacity = match ? '1' : '.4'; btn.style.cursor = match ? 'pointer' : 'not-allowed'; } function closeConfirm() { document.getElementById('confirm-modal').style.display = 'none'; _rollTarget = null; _confirmPhrase = ''; } async function executeRollover() { if (!_rollTarget) return; const yr = _rollTarget; document.getElementById('spin-confirm').classList.remove('hidden'); document.getElementById('confirm-btn').disabled = true; try { const resp = await fetch('/api/rollover', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ year_group: yr }), }); const d = await resp.json(); closeConfirm(); if (d.ok) { const msg = d.archived ? ('Y6 archived — ' + d.archived + ' pupils removed from active classes.') : ('Y' + yr + ' rolled — ' + d.moved + ' pupils moved to Y' + (parseInt(yr)+1) + ' (Pending).'); showStatus(msg); await loadCounts(); } else { showStatus('Rollover failed: ' + d.error, 'error'); } } catch(e) { closeConfirm(); showStatus('Request failed: ' + e.message, 'error'); } } loadCounts();
